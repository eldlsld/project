**event
create table event (
	eid number(6),
	image varchar2(50),
	title varchar2(100),
	content varchar2(2000),
	primary key (eid)
);

create sequence seq_event
increment by 1
start with 1
maxvalue 99999
nocycle
nocache

select seq_event.nextval from dual;
select seq_event.currval from dual;
